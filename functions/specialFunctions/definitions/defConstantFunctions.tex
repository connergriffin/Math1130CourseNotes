\begin{definition}
    A function, \(f,\) is a \emph{constant function} if there is a \(k\) such that for all real numbers \(x\) \(f\left(x\right)=k.\)
\end{definition}

\begin{question}
    Which of the following are constant functions?
    \begin{selectAll}
        \choice[correct]{\(f(x) = 2\)}
        \choice[correct]{\(g(x) = -1\)}
        \choice{\(h(t) = t-1\)}
        \choice{\(r(x) = \frac{1}{1-x}\)}
    \end{selectAll}
\end{question}
