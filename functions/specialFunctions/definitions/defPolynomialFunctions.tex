\begin{definition}
    A function, \(f,\) is a \emph{polynomial function} if there is a positive whole number \(n\) and real numbers \(a_0,a_1,\dots, a_n\) with \(a_n\ne 0\) such that for all real numbers \(x,\) \(f\left(x\right)=a_nx^n+a_{n-1}x^{n-1}+\dots+a_1x+a_0.\) The degree of \(f\) is \(n\) and the leading coefficient is \(a_n.\)
\end{definition}

The domain of any polynomial is the set of all real numbers.

\begin{question}
    Which of the following are polynomial functions?
    \begin{selectAll}
        \choice[correct]{\(f(x) = 2\)}
        \choice[correct]{\(g(x) = -1+x\)}
        \choice[correct]{\(h(t) = t^2+2t-1\)}
        \choice{\(r(x) = \frac{1}{1-x}\)}
        \choice[correct]{\(r(t) = \frac{4}{3}t^3-1\)}
    \end{selectAll}
\end{question}
