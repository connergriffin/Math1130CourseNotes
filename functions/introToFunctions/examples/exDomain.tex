\begin{question}
    Find the domain of the function \(f\left(x\right) =\frac{x+1}{x^2-3x+2}\)

    \begin{solution}
    We cannot have division by \(0.\) Thus \(x^2-3x+2\ne0. \)
        \begin{gather*}
            x^2-3x+2\ne0 \\
            (x-2)(x-1) \ne 0 \\
            x\ne 2 \ \textrm{and} \ x \ne 1
        \end{gather*}
            The domain of \(f\) is \(\left(-\infty,\infty\right) - \{1,2\}.\)
    \end{solution}
\end{question}

\begin{question}
    Find the domain of the function \(g\left(x\right) =\frac{1}{\sqrt{x+3}}\)

    \begin{solution}
    We cannot have division by \(0.\) Thus \(\sqrt{x+3}\ne0. \) Combine this with \(\sqrt{x+3}\ge 0.\) We get that \(\sqrt{x+3} >0.\)
        \begin{gather*}
            \sqrt{x+3}>0 \\
            x+3 > 0 \\
            x> -3
        \end{gather*}
            The domain of \(g\) is \(\left(-3,\infty\right).\)
    \end{solution}
\end{question}
