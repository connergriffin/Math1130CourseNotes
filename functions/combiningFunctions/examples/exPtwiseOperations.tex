\begin{example}
    If \(f(x) = \frac{1}{2}x^2 + 1\) and \(g(x)  = x^2 - x + 2\) find and simplify
    \begin{enumerate}
        \item \(\left(f+g\right)\left(x\right)\)
        \item \(\left(f-g\right)\left(x\right)\)
        \item \(\left(fg\right)\left(x\right)\)
        \item \(\left(\frac{f}{g}\right)\left(x\right)\)
        \item \(\left(\frac{2}{3}g\right)\left(x\right)\)
    \end{enumerate}
    \begin{solution}
        (a) \(\left(f+g\right)\left(x\right) = f(x)+g(x) = \frac{1}{2}x^2 + 1 + x^2 - x + 2 = \frac{3}{2}x^2-x+3\)

        (b) \(\left(f-g\right)\left(x\right) = f(x)-g(x) = \frac{1}{2}x^2 + 1 - \left(x^2 - x + 2\right) =  \frac{1}{2}x^2 + 1 - x^2 + x - 2 = -\frac{1}{2}x^2+x-1.\)

        (c) \(\left(fg\right)\left(x\right) = f(x)g(x) = \left(\frac{1}{2}x^2 + 1\right)\left(x^2 - x + 2\right) = \frac{1}{2}x^4 -\frac{1}{2}x^3+2x^2-x+2\)

        (d) \(\left(\frac{f}{g}\right)\left(x\right) = \frac{f(x)}{g(x)} = \frac{\frac{1}{2}x^2 + 1}{x^2 - x + 2}\)

        (e) \(\left(\frac{2}{3}g\right)\left(x\right) = \frac{2}{3}\left(x^2-x+2\right) = \frac{2}{3}x^2-\frac{2}{3}x+\frac{4}{3}\)
    \end{solution}
\end{example}
