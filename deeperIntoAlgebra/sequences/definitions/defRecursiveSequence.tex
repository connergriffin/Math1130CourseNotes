\begin{definition}
    An infinite sequence is recursive if \(a_1\) is given and for \(k\ge2\) the \(k\)th term is defined in terms of previous terms.
\end{definition}

\begin{example}
    Define an infinite sequence \(a\) by
    \[a_1 = 1 \ \textrm{and} \ a_k = ka_{k-1} \ \textrm{for} \ k\ge 2.\]
    Compute the first \(5\) terms.
    \begin{gather*}
        a_1 =1 \\
        a_2 = 2a_1 = 2 \\
        a_3 = 3a_2 = 3\left(2\right) = 6\\
        a_4 = 4a_3 = 4 \left(6\right) = 24\\
        a_5 = 5a_5 = 5\left(24\right) = 120
    \end{gather*}
    Recall that \(k!\) is the product of the first \(k\) positive integers and note that \(a_k = k!\) defines \(a.\) 
\end{example}
