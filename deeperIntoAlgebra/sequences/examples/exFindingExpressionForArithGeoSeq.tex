\begin{example}
    Given that \(\frac{3}{2}\) and \(3\) are the second and third terms of an arithmetic sequence, find an explicit formula for the sequence.
    \begin{solution}
        We need the common difference and the initial term.
        \begin{gather*}
        d = b_3 - b_2 = 3- \frac{3}{2} = \frac{3}{2}\\
        a = b_1 = b_2 - d = \frac{3}{2} - \frac{3}{2} =0.
        \end{gather*}
        Thus \(b_k = \left(k-1\right)\frac{3}{2}.\)
    \end{solution}
\end{example}
