\begin{example}
    List the first \(4\) terms of the sequence \(\left(a_{k}\right)_{k}^{\infty}\) whose \(k\)th term is given by \(a_k = k^2-k+6.\) What is the \(26\)th term?
    \begin{solution}
        \begin{gather*}
            a_1 = 1^2 - 1 + 6 = 6 \\
            a_2 = 2^2 - 2 + 6 = 8 \\
            a_3 = 3^2 - 3 + 6 = 12 \\
            a_4 = 4^2 - 4 + 6 = 18 \\
            a_26 = 26^2 - 26 + 6 = 656
        \end{gather*}
    \end{solution}
\end{example}
