\begin{definition}
    Every straight line is the graph of an equation of the form
    \[Ax+By+C = 0\]
    where \(A\) and \(B\) are not both \(0.\)
    This equation is called the \emph{general linear equation.}
\end{definition}

\begin{example}
    Rearrange the equation \(3(y-2)=5(x+7)\) into the general linear form.

    \begin{solution}
        \begin{gather*}
            3(y-2)=5(x+7) \\
            3y-6=5x+35 \\
            -5x+3y-6-35=0\\
            -5x+3y-41 =0.
        \end{gather*}
    \end{solution}
\end{example}
