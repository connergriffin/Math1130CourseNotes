\begin{definition}
    Two lines are \emph{parallel} if they have the same slope.
\end{definition}

Every line is parallel to itself.

\begin{definition}
    Two lines with slopes \(m_0\) and \(m_1\) are \emph{perpendicular} if
    \[m_0 = -\frac{1}{m_1}.\]
\end{definition}

\begin{example}
    Determine if the two lines are parallel or perpendicular.

    \[y-2x-3=0 \ \textrm{and} \ 2y-4x+5 =0\]
    \begin{multipleChoice}
        \choice[correct] {parallel}
        \choice {perpendicular}
        \choice {neither}
    \end{multipleChoice}

    \[y-2x-3=0 \ \textrm{and} \ 4y+2x+1 =0\]
    \begin{multipleChoice}
        \choice{parallel}
        \choice[correct]{perpendicular}
        \choice {neither}
    \end{multipleChoice}

    \[y-2x-3=0 \ \textrm{and} \ y-1x+3 =0\]
    \begin{multipleChoice}
        \choice {parallel}
        \choice {perpendicular}
        \choice[correct] {neither}
    \end{multipleChoice}

    \[x=3 \ \textrm{and} \ y =-5\]
    \begin{multipleChoice}
        \choice{parallel}
        \choice[correct]{perpendicular}
        \choice{neither}
    \end{multipleChoice}
\end{example}
