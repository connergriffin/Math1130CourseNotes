\begin{definition}
    A \emph{quadratic form equation} in the variable \(x\) is an equation that can be written in the form
    \[ax^{2k}+bx^k+c=0\]
    where \(a,b,c,k\) are real numbers and \(a\ne 0\) and \(k\ne 0.\)
\end{definition}

\begin{question}
    Which of the following are quadratic form equations?
    \begin{selectAll}
        \choice[correct]{\(x^2+x+1=0\)}
        \choice[correct]{\(x^6+3x^3+2=0\)}
        \choice[correct]{\(\frac{1}{y^2}+3\frac{1}{y} = 7\)}
        \choice{\(t^4+t^3=1\)}
    \end{selectAll}
\end{question}
