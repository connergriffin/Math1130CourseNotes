\begin{example}
    Find the present value of an annuity of \(\$1500\) per month for \(2\) years at an interest rate of \(4.25\%\) compounded monthly.

    \begin{solution}
        \(R = \$1500.\) We need the periodic rate \(r\) and the number of payments \(n.\) There are \(12\) interest periods in a year and thus
        \[r = \frac{0.0425}{12} = 0.00354 \ \textrm{and} \ n = 2\cdot 12 = 24.\]
        The present value of the annuity is \(1500a\left(24,0.00354\right) = 1500\frac{1-\left(1+0.00354\right)^{-24}}{0.00354} \approx 34454.73\)
    \end{solution}
\end{example}

\begin{example}
    Suppose you buy a vehicle for \(\$12000\) which you will pay off in \(48\) months at \(7\%\) compounding monthly. Find the amount of each monthly payment. How much is paid in interest?

    \begin{solution}
        The total present value is \(\$12000,\) \(r = \frac{0.07}{12}\approx 0.0058,\) \(n=48.\)
        \[12000 = R\frac{1-\left(1.0058\right)^{-48}}{.0058}.\]
        We need to solve for \(R.\)

        \[\frac{1-\left(1.0058\right)^{-48}}{.0058} \approx 41.793\]

        Thus \(R \approx \frac{12000}{41.793} = 287.13.\)

        Total amount paid in interest is \(287.13\cdot 48 -12000 = 1782.24.\)
    \end{solution}
\end{example}
