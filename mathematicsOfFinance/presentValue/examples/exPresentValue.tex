\begin{example}
    Find the present value of the given future payment at the specified interest rate.
    
    \(\$2500\) due in \(6\) years at \(7.5\%\) compounded monthly.

    \begin{solution}
        \(S=2500,\) \(r = \frac{0.075}{12} = 0.00625,\) \(n = 6\cdot 12 = 36.\)
        The present value is \(P = 2500\left(1.00625\right)^{-36} = 1997.69\)
    \end{solution}
\end{example}
