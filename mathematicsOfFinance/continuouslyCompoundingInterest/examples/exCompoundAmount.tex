\begin{example}
    If \(\$1000\) is invested at an annual rate of \(3\%\) compounded continuously, find the compound amount at the end of
    \begin{enumerate}
        \item \(1\) year
        \item \(5\) years
    \end{enumerate}
    \begin{solution}
        \((a)\) \(S = 1000e^{0.03} \approx 1030.45\)

        \((b)\) \(S = 1000e^{0.03*5} \approx 1161.83\)
    \end{solution}
\end{example}
